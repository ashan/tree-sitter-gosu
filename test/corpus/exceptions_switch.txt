Exceptions & Switch
==================

try {
  throw new Exception()
} catch (e : Exception) {
  print(e)
} finally {
  cleanup()
}

switch (x) {
  case 1:
    print("one")
    break
  default:
    print("other")
}

do {
  x++
} while(x < 10)
